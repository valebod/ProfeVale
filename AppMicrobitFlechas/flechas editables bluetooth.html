<head>

  <meta charset="UTF-8">

  <title>Control Micro:bit</title>

  <script>

    let device, server, uartService, txCharacteristic;

    let commands = { up: "U", down: "D", left: "L", right: "R" };


    async function connect() {

      try {

        device = await navigator.bluetooth.requestDevice({

          filters: [{ namePrefix: "BBC micro:bit" }],

          optionalServices: ["6e400001-b5a3-f393-e0a9-e50e24dcca9e"] // Nordic UART Service

        });

        server = await device.gatt.connect();

        uartService = await server.getPrimaryService("6e400001-b5a3-f393-e0a9-e50e24dcca9e");

        txCharacteristic = await uartService.getCharacteristic("6e400002-b5a3-f393-e0a9-e50e24dcca9e");

        alert("Conectado a la micro:bit");

      } catch (error) {

        console.error(error);

      }

    }


    async function sendCommand(cmd) {

      if (!txCharacteristic) return alert("Primero conectate a la micro:bit");

      let encoder = new TextEncoder();

      await txCharacteristic.writeValue(encoder.encode(cmd));

    }


    function updateCommand(dir) {

      const value = document.getElementById(dir+"Cmd").value;

      commands[dir] = value;

    }

  </script>

  <script src="https://cdn.tailwindcss.com"></script>

</head>

<body class="flex flex-col items-center p-6 bg-gray-100 min-h-screen">

  <h1 class="text-2xl font-bold mb-4">Control Bluetooth Micro:bit</h1>

  <button onclick="connect()" class="mb-6 px-4 py-2 bg-blue-500 text-white rounded">Conectar</button>


  <div class="grid grid-cols-3 gap-4 mb-8">

    <div></div>

    <button onclick="sendCommand(commands.up)" class="p-4 bg-green-400 rounded">⬆️</button>

    <div></div>

    <button onclick="sendCommand(commands.left)" class="p-4 bg-green-400 rounded">⬅️</button>

    <div></div>

    <button onclick="sendCommand(commands.right)" class="p-4 bg-green-400 rounded">➡️</button>

    <div></div>

    <button onclick="sendCommand(commands.down)" class="p-4 bg-green-400 rounded">⬇️</button>

    <div></div>

  </div>


  <h2 class="text-xl font-semibold mb-2">Configuración</h2>

  <div class="grid grid-cols-2 gap-2">

    <label>Arriba: <input id="upCmd" value="U" onchange="updateCommand('up')" class="border p-1"></label>

    <label>Abajo: <input id="downCmd" value="D" onchange="updateCommand('down')" class="border p-1"></label>

    <label>Izquierda: <input id="leftCmd" value="L" onchange="updateCommand('left')" class="border p-1"></label>

    <label>Derecha: <input id="rightCmd" value="R" onchange="updateCommand('right')" class="border p-1"></label>

  </div>
